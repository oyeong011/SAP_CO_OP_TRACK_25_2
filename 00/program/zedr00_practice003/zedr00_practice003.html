FORM SCREEN_SETTING .
  LOOP AT SCREEN.
    IF SCREEN-GROUP1 = C_M2.
      IF P_R1 = C_X.
        SCREEN-ACTIVE = C_0.
      ELSEIF P_R2 = C_X.
        SCREEN-ACTIVE = C_1.
      ENDIF.
    ELSEIF SCREEN-GROUP1 = C_M1.
      IF P_R1 = C_X.
        SCREEN-ACTIVE = C_1.
      ELSEIF P_R2 = C_X.
        SCREEN-ACTIVE = C_0.
      ENDIF.
    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.
