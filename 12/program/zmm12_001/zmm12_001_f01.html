FORM SEARCH_VENDOR .
   CLEAR : GS_ALV_OUTPUT, GT_ALV_OUTPUT.

  SELECT SINGLE
      A~LIFNR   "####
      A~NAME1   " ####
      A~LAND1   " ###
      A~KTOKK " #####
      A~STCD1   " ####
      A~STCD2   " #####
      A~STRAS   " ##
      B~BUKRS  "####
      B~LOEVM  " #####
      B~AKONT  " ##
      B~ZTERM  " ####
      C~EKORG  " ####
      C~EKGRP  " ####
      C~WAERS  " ######
      C~MWSKZ  " ####
    INTO CORRESPONDING FIELDS OF GS_ALV_OUTPUT
    FROM ZLFA1_12 AS A
      INNER JOIN ZLFB1_12 AS B
      ON A~LIFNR = B~LIFNR
      INNER JOIN ZLFM1_12 AS C
      ON A~LIFNR = C~LIFNR
    WHERE A~LIFNR = P_LIFNR
      AND B~BUKRS = P_BUKRS.

  APPEND GS_ALV_OUTPUT TO GT_ALV_OUTPUT.
ENDFORM.
