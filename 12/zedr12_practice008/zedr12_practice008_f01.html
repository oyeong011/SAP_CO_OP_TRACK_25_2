FORM SET_SCREEN .

  IF SCREEN-GROUP1 = 'M1'.

      SCREEN-ACTIVE = '1'.
    ELSEIF P_R2 = 'X'.
      SCREEN-ACTIVE = '0'.
    ENDIF.

      SCREEN-ACTIVE = '1'.
    ELSEIF P_R1 = 'X'.
      SCREEN-ACTIVE = '0'.
    ENDIF.
  ENDIF.
  MODIFY SCREEN.
ENDLOOP.
ENDFORM.
