FORM GET_DATA .
  SELECT A~ZCODE
         A~ZPERNR
         A~ZKNAME
         A~ZENAME
         A~ZGENDER
         B~ZMAJOR
         B~ZMNAME
    FROM ZEDT25_001 AS A
    INNER JOIN ZEDT25_002 AS B
    ON A~ZCODE = B~ZCODE
    INTO CORRESPONDING FIELDS OF TABLE GT_STUDENT
    WHERE A~ZCODE IN S_ZCODE.
