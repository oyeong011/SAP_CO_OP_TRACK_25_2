FORM GET_DATA_EMPLOYEE .
  RANGES : LR_DEPCODE FOR ZEDT24_102-ZDEPCODE,
           LR_QUIT FOR ZEDT24_102-ZQFLAG.

  ENDIF.

  SELECT * FROM ZEDT24_102 AS 102
      INNER JOIN ZEDT24_103 AS 103 ON 102~ZPERNR = 103~ZPERNR
      INNER JOIN ZEDT24_106 AS 106 ON 102~ZPERNR = 106~ZPERNR

FORM MODIFY_DATA_EMPLOYEE .
  LOOP AT GT_IN_EMPLOYEE INTO GS_IN_EMPLOYEE.
    CLEAR GS_OUT_EMPLOYEE.

    MOVE-CORRESPONDING GS_IN_EMPLOYEE TO GS_OUT_EMPLOYEE.
    PERFORM ZDEPCODE_2_ZDEPNAME USING GS_IN_EMPLOYEE-ZDEPCODE GS_OUT_EMPLOYEE-ZDEPNAME.
    PERFORM ZDEPRANK_2_ZDEPRANKNAME USING GS_IN_EMPLOYEE-ZDEPRANK GS_OUT_EMPLOYEE-ZDEPRANKNAME.
    PERFORM ZQFLAG_2_ZQSTR USING GS_IN_EMPLOYEE-ZQFLAG GS_OUT_EMPLOYEE-ZQSTR.
    PERFORM ZGENDER_2_ZGENDERSTR USING GS_IN_EMPLOYEE-ZGENDER GS_OUT_EMPLOYEE-ZGENDERSTR.
    PERFORM ZBANKCODE_2_ZBANKNAME USING GS_IN_EMPLOYEE-ZBANKCODE GS_OUT_EMPLOYEE-ZBANKNAME.

    APPEND GS_OUT_EMPLOYEE TO GT_OUT_EMPLOYEE.
  ENDLOOP.
ENDFORM.
