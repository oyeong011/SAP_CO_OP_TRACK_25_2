FORM MODIFY_SELECTION_SCREEN .
  LOOP AT SCREEN.

    " ####
    IF P_R1 = 'X'.
      IF SCREEN-GROUP1 = 'M1'.
        SCREEN-ACTIVE = 1.
      ELSEIF SCREEN-GROUP1 = 'M2'.
        SCREEN-ACTIVE = 0.
      ENDIF.

    " ####
    ELSEIF P_R2 = 'X'.
      IF SCREEN-GROUP1 = 'M1'.
        SCREEN-ACTIVE = 0.
      ELSEIF SCREEN-GROUP1 = 'M2'.
        SCREEN-ACTIVE = 1.
      ENDIF.
    ENDIF.

    MODIFY SCREEN.
  ENDLOOP.
ENDFORM.
