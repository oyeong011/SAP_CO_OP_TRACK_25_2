FORM REMOVE_RBKP.

  LOOP AT GT_RBKP_ITAB INTO DATA(LINE).
    IF LINE-ZCHECK EQ 'X'.

      SELECT *
        INTO CORRESPONDING FIELDS OF LS_RBKP
        FROM Z23RBKP
