FORM GET_DATE.
  DATA :LV_YEAR TYPE I,
        LV_MONTH TYPE I,
        LV_NEXT_MONTH TYPE SY-DATUM,
        LV_TODAY TYPE SY-DATUM.

  " ### ##+# + 01
  ZSTART_DAY = SY-DATUM.
  ZSTART_DAY+6(2) = '01'.

  " ### ## ### #
  LV_TODAY = SY-DATUM.
  LV_YEAR = LV_TODAY+0(4).
  LV_MONTH = LV_TODAY+4(2).

  IF LV_MONTH = 12.
    LV_NEXT_MONTH = |{ LV_YEAR + 1 }0101 |.
  ELSE.
    LV_NEXT_MONTH = |{ LV_YEAR }{ LV_MONTH + 1 WIDTH = 2 PAD = '0' }01 |.
  ENDIF.

  ZEND_DAY = LV_NEXT_MONTH - 1.

ENDFORM.
