FORM MODIFY_DATA .

    " ### ##
    GS_OUT-FCURR = GS_EXCEL-FCURR.

    " #### M ##, ###### ## ### ##
    GS_OUT-KURST = 'M'.
    GS_OUT-GDATU = P_DATE.

    " JPY, VND# ### 100## ##
    CASE GS_OUT-FCURR.
      WHEN 'JPY' OR 'VND'.
        GS_OUT-FFACT = 100.
      WHEN OTHERS.
        GS_OUT-FFACT = 1.
    ENDCASE.
    GS_OUT-TFACT = 1.

    " ### ## ###, ###
    GS_OUT-UNAME = SY-UNAME.
    GS_OUT-DATUM = SY-DATUM.

    APPEND GS_OUT TO GT_OUT.
  ENDLOOP.
