    INTO LV_CHECK_EBELN

    MESSAGE '#### ## #### #####.' TYPE 'I'.

      INTO LV_CHECK_WERKS
      FROM ZEDT16_EKPO
      WHERE EBELN = P_EBELN1
        AND WERKS = P_WERKS1.

      MESSAGE '## ##### #### ## ######.' TYPE 'I'.
      STOP.
    ENDIF.
