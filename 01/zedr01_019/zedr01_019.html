FORM MODIFY_DATA .

  CLEAR GS_STUDENT.

  LOOP AT GT_STUDENT INTO GS_STUDENT.
    IF GS_STUDENT-ZGENDER = 'M'.
      GS_STUDENT-ZCOLOR = '@0A@'.
    ELSE.
      GS_STUDENT-ZCOLOR = '@08@'.
    ENDIF.

    MODIFY GT_STUDENT FROM GS_STUDENT INDEX SY-TABIX.
    CLEAR GS_STUDENT.
  ENDLOOP.
ENDFORM.
