FORM modify_data.
  CLEAR gs_student.

  LOOP AT gt_student INTO gs_student.
    IF gs_student-zename IS NOT INITIAL.
      gs_student-zcolor = '@0A@'.
    ENDIF.

    MODIFY gt_student FROM gs_student INDEX sy-tabix.

  ENDLOOP.
ENDFORM.
