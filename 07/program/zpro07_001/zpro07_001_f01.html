FORM GET_DATA .
  SELECT a~lifnr a~name1 a~stcd2 a~stras a~land1 b~bukrs
  INTO CORRESPONDING FIELDS OF TABLE gt_search
  FROM zpro07_001 AS a
  INNER JOIN zpro07_002 AS b
    ON a~lifnr = b~lifnr
  WHERE a~lifnr = p_lifnr
    AND b~bukrs = p_bukrs.
