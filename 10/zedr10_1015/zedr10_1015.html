FORM MODIFY_DATA .
  CLEAR GS_STUDENT.

  LOOP AT GT_STUDENT INTO GS_STUDENT.
    IF GS_STUDENT-ZGRADE = 'F'.
      GS_STUDENT-ZCOLOR = '@0A@'.
