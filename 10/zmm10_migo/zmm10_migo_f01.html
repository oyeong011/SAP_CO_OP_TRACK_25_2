       K~LIFNR K~BEDAT K~WAERS

      MESSAGE '## ### ### #######.' TYPE 'S'.
    ENDIF.

    "====================================================================
    " CASE 2: ## ## ## (#### ##)
    "====================================================================
  ELSEIF R_DISP = 'X'.

    " 1. ## #### ##
*    IF P_MBLNR IS INITIAL.
*      MESSAGE '## ## ### #####.' TYPE 'S' DISPLAY LIKE 'E'.
*      EXIT.
*    ENDIF.

    SELECT A~MBLNR A~MJAHR A~BUDAT AS BEDAT " ### #### BEDAT ### ##
           A~USNAM A~BLART

    AND P~WERKS = P_WERKS
      AND B~BWART = '101'.
    .
*      WHERE A~MBLNR = P_MBLNR

    ENDIF.

      "      MSEG## #### #### PO ##### ## ##### ##
      SELECT SINGLE MAKTX INTO GS_ITEM-MAKTX
        FROM ZEDT10_EKPO
        WHERE MATNR = GS_ITEM-MATNR
          AND EBELN = GS_ITEM-EBELN
          AND EBELP = GS_ITEM-EBELP.

      " PO #### ### ## ###(ZMARA10)## ### (####)

      GS_ITEM-MESSAGE = '## ##'.

    ENDLOOP.

  ENDIF.

ENDFORM.

FORM DISPLAY_ALV .
  IF GO_CONTAINER IS INITIAL.
    CREATE OBJECT GO_CONTAINER
      EXPORTING
        CONTAINER_NAME = 'CC_ALV'.

    CREATE OBJECT GO_ALV
      EXPORTING
        I_PARENT = GO_CONTAINER.

    PERFORM SET_LAYOUT.
    PERFORM SET_FCAT.

    CALL METHOD GO_ALV->SET_TABLE_FOR_FIRST_DISPLAY
      EXPORTING
        IS_LAYOUT       = GS_LAYOUT
      CHANGING
        IT_OUTTAB       = GT_ITEM
        IT_FIELDCATALOG = GT_FCAT.
  ELSE.
    PERFORM REFRESH_ALV.
  ENDIF.
ENDFORM.
