FORM CONTROL_SCREEN .

  LOOP AT SCREEN.
    IF P_CREATE = 'X'.
      IF SCREEN-GROUP1 = 'M1'.
        SCREEN-ACTIVE = 1.
      ELSEIF SCREEN-GROUP1 = 'M2'.
        SCREEN-ACTIVE = 0.
      ENDIF.
    ELSEIF P_FIND = 'X'.
      IF SCREEN-GROUP1 = 'M2'.
        SCREEN-ACTIVE = 1.
      ELSEIF SCREEN-GROUP1 = 'M1'.
        SCREEN-ACTIVE = 0.
      ENDIF.
    ENDIF.

    MODIFY SCREEN.
  ENDLOOP.

ENDFORM.
