        SCREEN-INPUT = 0.
        SCREEN-ACTIVE = 1.
        GV_SEL_TYPE = 'GR'.

        " PO# ### #### #### (PO ## ##)

        SCREEN-ACTIVE = 0.
      ENDIF.
    ENDIF.

  ENDLOOP.

  PERFORM DISPLAY_ALV.
