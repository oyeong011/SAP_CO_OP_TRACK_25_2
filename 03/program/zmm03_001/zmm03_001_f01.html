FORM CONTROL_SCREEN .
  LOOP AT SCREEN.
    IF SCREEN-GROUP1 = 'CRE'.
      IF P_CRE = 'X'.
        SCREEN-ACTIVE = '1'.
      ELSE.
        SCREEN-ACTIVE = '0'.
      ENDIF.
    ELSEIF SCREEN-GROUP1 = 'DSP'.
      IF P_DISP = 'X'.
        SCREEN-ACTIVE = '1'.
      ELSE.
        SCREEN-ACTIVE = '0'.
      ENDIF.
    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.
ENDFORM.

FORM GET_DATA .

           INNER JOIN ZEDT03_LFB1 AS B ON A~LIFNR = B~LIFNR

FORM VIEW_EDIT .
  LOOP AT SCREEN.
    IF P_CRE = 'X'.
      IF SCREEN-NAME = 'GV_LIFNR' OR SCREEN-NAME = 'GV_KTOKK' OR SCREEN-NAME = 'GV_BUKRS'.
        SCREEN-INPUT = 0.
      ENDIF.
    ELSE. "### #
      IF SCREEN-NAME = 'GV_STCD2' OR SCREEN-NAME = 'GV_SAVE'.
        SCREEN-INPUT = 1.
      ELSE.
        SCREEN-INPUT = 0.
      ENDIF.
    ENDIF.


    IF SCREEN-NAME = 'GV_STCD1' OR SCREEN-NAME = 'GV_STCD1_TXT'.
      SCREEN-ACTIVE = 0.
    ENDIF.

    CASE GV_KTOKK.
        WHEN '2000'. "##

            SCREEN-ACTIVE = 0.
          ENDIF.

            SCREEN-ACTIVE = 0.
          ENDIF.
        ENDCASE.
    MODIFY SCREEN.
