FORM CHECK_MATNR USING PV_MATNR TYPE ZEKPO_22-MATNR
                 CHANGING PV_VALID TYPE C.

  DATA : LV_COUNT TYPE I.

  CLEAR PV_VALID.

  SELECT COUNT(*)
    INTO LV_COUNT
    FROM ZMARA_22
    WHERE ZMATNR = PV_MATNR.

  IF LV_COUNT > 0.
    PV_VALID = 'S'.
  ELSE.
    PV_VALID = 'E'.
  ENDIF.

ENDFORM.
